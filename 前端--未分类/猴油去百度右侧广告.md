### 方式1 自己手写
```
 //===========猴油去百度右侧广告的开始======================
 // requestAnimationFrame 比 setInterval好，不会出现卡顿的问题
    let timer = requestAnimationFrame(timerfn);
    function timerfn(){
         if(document.getElementById("content_right")){
             document.getElementById("content_right").style.display = "none"
             cancelAnimationFrame(timer);
         }else{
             timer = requestAnimationFrame(timerfn);

         }
    }
//===============猴油去百度右侧广告的结束======================
```
