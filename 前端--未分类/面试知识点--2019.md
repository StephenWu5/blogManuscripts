#### 算法（排序）

```js
var Sort=function(arr){  // 冒泡排序
    for(var i=0;i<arr、length-1;i++){ 
        for(var j=i+1;j<arr、length;j++){ 
            if(arr[i]>arr[j]){/*如果前面的数据比后面的大就交换位置*/
                var list=arr[i]; 
                arr[i]=arr[j]; 
                arr[j]=list; 
            }  
        } 
    }  
    return arr; 
} 
```
